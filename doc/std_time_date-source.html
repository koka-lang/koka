// An ISO day of the week, starting at Monday (=1) up to Sunday (=7).
public type weekdaystd/time/date/weekday: V {
  Monweekday: weekday
  Tueweekday: weekday
  Wedweekday: weekday
  Thuweekday: weekday
  Friweekday: weekday
  Satweekday: weekday
  Sunweekday: weekday
}

// Convert a `:weekday` to an `:int` using the ISO definition which starts at Monday as 1,
// up to Sunday as 7.
public fun intstd/time/date/int: (wd : weekday) -> int( wdwd: weekday : weekdaystd/time/date/weekday: V ) : intstd/core/types/int: V {
  match(wdwd: weekday) {
